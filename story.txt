using UnityEngine;
using UnityEngine.UI;

public class QuizManager : MonoBehaviour
{
    public Text questionText;
    public Button[] optionButtons;

    private string question = "What is the powerhouse of the cell?";
    private string[] options = { "Ribosome", "Mitochondria", "Nucleus", "Chloroplast" };
    private int correctIndex = 1;

    void Start()
    {
        questionText.text = question;
        for (int i = 0; i < optionButtons.Length; i++)
        {
            int index = i; // capture index
            optionButtons[i].GetComponentInChildren<Text>().text = options[i];
            optionButtons[i].onClick.AddListener(() => CheckAnswer(index));
        }
    }

    void CheckAnswer(int selected)
    {
        if (selected == correctIndex)
            Debug.Log("Correct!");
        else
            Debug.Log("Wrong Answer!");
    }
}
